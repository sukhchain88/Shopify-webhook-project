# Render.com Environment Variables Template
# Copy these variables to your Render.com service environment variables

# Node Environment
NODE_ENV=production

# Database Configuration (Render.com PostgreSQL)
# Use either DATABASE_URL (recommended for Render.com) OR individual DB variables
DATABASE_URL=postgresql://username:password@hostname:port/database_name

# Alternative individual database variables (if not using DATABASE_URL)
# DB_NAME=your_database_name
# DB_USER=your_database_user
# DB_PASSWORD=your_database_password
# DB_HOST=your_database_host
# DB_PORT=5432

# Shopify Configuration
SHOPIFY_SHOP_DOMAIN=your-shop.myshopify.com
SHOPIFY_ACCESS_TOKEN=your_shopify_access_token
SHOPIFY_API_VERSION=2023-10

# Server Configuration
PORT=3000

# Webhook Configuration
WEBHOOK_BASE_URL=https://your-app-name.onrender.com

# Instructions for Render.com:
# 1. Go to your Render.com dashboard
# 2. Select your web service
# 3. Go to Environment tab
# 4. Add each variable above (without the # comments)
# 5. For DATABASE_URL, use the Internal Database URL from your PostgreSQL service
# 6. Make sure NODE_ENV is set to "production"
# 7. Set PORT to 3000 (or leave empty, Render will set it automatically)
